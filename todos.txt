    avatar_id serial PRIMARY KEY,
    username VARCHAR(50) UNIQUE,
    avatar_character VARCHAR(50),
    xp INTEGER,

	health INTEGER,
    user_id INTEGER UNIQUE,
    FOREIGN KEY (user_id) REFERENCES users(id)

    wpm FLOAT NOT NULL,

	avatar_id INTEGER NOT NULL,
